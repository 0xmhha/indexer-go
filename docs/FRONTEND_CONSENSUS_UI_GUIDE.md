### 4.2 Key Component Implementations

#### Address Component

```tsx
// components/common/Address.tsx
import { useState } from 'react';
import { CopyIcon, ExternalLinkIcon, CheckIcon } from 'lucide-react';

interface AddressProps {
  address: string;
  truncate?: boolean;
  showCopy?: boolean;
  showExplorer?: boolean;
  label?: string;
  variant?: 'default' | 'proposer' | 'validator';
}

export function Address({
  address,
  truncate = true,
  showCopy = true,
  showExplorer = true,
  label,
  variant = 'default'
}: AddressProps) {
  const [copied, setCopied] = useState(false);

  const displayAddress = truncate
    ? `${address.slice(0, 6)}...${address.slice(-4)}`
    : address;

  const handleCopy = async () => {
    await navigator.clipboard.writeText(address);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };

  return (
    <div className={`address-component ${variant}`}>
      {label && <span className="address-label">{label}</span>}
      <span className="address-value" title={address}>
        {displayAddress}
      </span>
      {showCopy && (
        <button onClick={handleCopy} className="address-action">
          {copied ? <CheckIcon size={14} /> : <CopyIcon size={14} />}
        </button>
      )}
      {showExplorer && (
        <a
          href={`/validators/${address}`}
          className="address-action"
        >
          <ExternalLinkIcon size={14} />
        </a>
      )}
    </div>
  );
}
```

#### Round Indicator Component

```tsx
// components/consensus/RoundIndicator.tsx
interface RoundIndicatorProps {
  round: number;
  size?: 'sm' | 'md' | 'lg';
  showLabel?: boolean;
}

export function RoundIndicator({
  round,
  size = 'md',
  showLabel = true
}: RoundIndicatorProps) {
  const getStatus = () => {
    if (round === 0) return { color: 'success', label: 'Success' };
    if (round === 1) return { color: 'warning', label: 'Round Change' };
    if (round === 2) return { color: 'orange', label: 'Multiple Rounds' };
    return { color: 'danger', label: 'Significant Delay' };
  };

  const status = getStatus();

  return (
    <div className={`round-indicator ${status.color} ${size}`}>
      <span className="round-value">R:{round}</span>
      {showLabel && <span className="round-label">{status.label}</span>}
    </div>
  );
}
```
